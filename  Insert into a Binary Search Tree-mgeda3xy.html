Insert into a Binary Search Tree

time:O(log2N)

---------
 TreeNode* insertIntoBST(TreeNode* root, int val) {
        if(root==NULL)
        {
            return new TreeNode(val);
        }
       TreeNode* root1=root;
        while(root1)
        {
            if(root1->valright!=NULL)
                {
                    root1=root1->right;
                }
                else{
                    root1->right=new TreeNode(val);
                    break;
                }
            }
            else{
                if(root1->left!=NULL)
                {
                    root1=root1->left;
                }
                else{
                    root1->left=new TreeNode(val);
                    break;
                }
            }
        }
        return root;
    }
