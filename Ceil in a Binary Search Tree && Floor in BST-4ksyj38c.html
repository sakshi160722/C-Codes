int findCeil(BinaryTreeNode *node, int x){
	int c=-1;
   while(node)
   {
	   if(node->data==x)
	   {
		   c=node->data;
		   return c;
	   }
	   else{
		   if(node->data>x)
		   {
			   c=node->data;
			   node=node->left;
		   }
		   else{
			    node=node->right;
		   }
		  
	   }
   }
	
	return c;

int floorInBST(TreeNode * root, int X)
{
	int c=-1;
	while(root)
	{
		if(root->val==X)
		{
			return root->val;
		}
		else{
			if(root->val>X)
			{
				root=root->left;
			}
			else{
				c=root->val;
				root=root->right;
			}
		}
	}
	return c;
}
