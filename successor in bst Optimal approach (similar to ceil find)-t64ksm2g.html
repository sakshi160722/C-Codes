successor in bst Optimal approach (similar to ceil find)

Time:O(h)
space: O(1)

 Node * inOrderSuccessor(Node *root, Node *x)
    {
          Node *c=NULL;
        if(root==NULL)
        {
            return c;
        }
  
    while(root)
    {
        if(root->data>x->data)
        {
            c=root;
            root=root->left;
        }
        else{
            root=root->right;
        }
    }
    return c;
    }
