Construct a Binary Tree from Preorder and Inorder Traversal 

Time Complexity:O(n)
Space: O(n)

 TreeNode* buildTree(vector& preorder, vector& inorder) {
        mapmp;
        for(int i=0;i& preorder,int prestart,int preEnd,vector& inorder,int instart,int inEnd,  map&mp)
     {
         if(prestart>preEnd || instart>inEnd)
         {
             return NULL;
         }
         TreeNode* node=new TreeNode(preorder[prestart]);
         int inroot=mp[node->val];
         int left=inroot-instart;
         node->left=solve(preorder,prestart+1,prestart+left,inorder,instart,inroot-1,mp);
         node->right=solve(preorder,prestart+left+1,preEnd,inorder,inroot+1,inEnd,mp);
         return node;
     }
