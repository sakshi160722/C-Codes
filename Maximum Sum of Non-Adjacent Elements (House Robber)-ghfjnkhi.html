  int solve(int n,vector& nums, vector&dp)
    {
        if(n==0)
        {
            return nums[0];
        }
        if(n<0)
        {
            return 0;
        }
        if(n==1)
        {
            return max(nums[0],nums[1]);
        }
        if(dp[n]!=-1)
        {
            return dp[n];
        }
        int p=nums[n]+solve(n-2,nums,dp);
        int np=0+solve(n-1,nums,dp);
        return dp[n]= max(p,np);
    }
    int rob(vector& nums) {
       
        int n=nums.size();
         vectordp(n+1,-1);
        return solve(n-1,nums,dp);
    }
