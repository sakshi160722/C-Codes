bool solve1(TreeNode* A, vector<int>&ans,int B)
{
    if(!A)
    {
        return false;
    }
    ans.push_back(A->val);
    if(A->val==B)
    {
        return true;
    }
    if(solve1(A->left,ans,B)||
    solve1(A->right,ans,B))
    return true;
    ans.pop_back();
    return false;
}

vector<int> Solution::solve(TreeNode* A, int B) {
    vector<int>ans;
    if(!A)
    {
        return ans;
    }
    solve1(A,ans,B);
    return ans;
}
