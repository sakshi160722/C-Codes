       stackst;

            TreeNode* node=st.top();

            if(node->right)
            {
                st.push(node->right);
            }
            if(node->left)

 void flatten(Node *root)
    {
        Node* cur=root;
             while(cur!=NULL)
             {
                 if(cur->left!=NULL)
                 {
                     Node* prev=cur->left;
                 
                 while(prev->right)
                 {
                     prev=prev->right;
                 }
                 prev->right=cur->right;
                 cur->right=cur->left;
                 cur->left=NULL;
                 }
                 cur=cur->right;
             }
    }
