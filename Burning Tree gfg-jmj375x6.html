      while(!q.empty())
      {
          Node* src=q.front();
          q.pop();
          if(src->data==target)
          {
              res=src;
          }
          if(src->left)
          {
              q.push(src->left);

          }
           if(src->right)
          {
              q.push(src->right);
              mp[src->right]=src;
          }
      }
      return res;
  }
