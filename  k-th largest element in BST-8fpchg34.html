k-th largest element in BST

better:
time:O(n)
space:O(n)

just use reverse of inorder --------------- right root left

 void solve(Node* root,vector&v)
    {
        if(!root)
        {
            return;
        }
        solve(root->right,v);
        v.push_back(root->data);
        solve(root->left,v);
    }
    int kthLargest(Node *root, int K)
    {
        vectorv;
        solve(root,v);
        int t=v[K-1];
        if(K>v.size())
        {
            return -1;
        }
        return t;
    }
