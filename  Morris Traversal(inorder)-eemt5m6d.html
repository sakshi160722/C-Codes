Morris Traversal(inorder)

 vector inorderTraversal(TreeNode *root)
    {
      vectorres;
        TreeNode* node=root;
        while(node)
        {
            if(node->left==NULL)
            {
                res.push_back(node->val);
                node=node->right;
            }
            else{
                TreeNode* prev=node->left;
                while(prev->right!=NULL && prev->right!=node)
                {
                    prev=prev->right;
                }
                if(prev->right==NULL)
                {
                    prev->right=node;
                    node=node->left;   
                }
                else{
                    prev->right=NULL;
                    res.push_back(node->val);
                    node=node->right;
                }
            }
        }
        return res;
    }
