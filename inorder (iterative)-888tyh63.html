inorder (iterative)

left root right
steps:1: make vector (ans) and stack(st)
2: assign cur=root  and check if cur !null so push to stack and move cur to left 
3: if there is no left of cur so firstly check (if st is empty) if yes ? then break: 
4:otherwise assign st top to cur and push cur to ans vector and after pop that element from stack 
5: and move cur to cur ->right
6:return ans;

 vector inorderTraversal(TreeNode *root)
    {
       vectorans;
        if(!root)
            return ans;
        stackst;
        TreeNode* cur=root;
        while(true)
        {
            if(cur!=NULL)
            {
                st.push(cur);
                cur=cur->left;
            }
            else{
                if(st.empty())
                {
                    break;
                }
                cur=st.top();
                st.pop();
                ans.push_back(cur->val);
                cur=cur->right;
            }
        }
        return ans;
    }
