void inorder(BinaryTreeNode *root,vector&v1)
{
	if(!root)
	{
		return;
	}
	inorder(root->left,v1);
	v1.push_back(root->data);
	inorder(root->right,v1);
}
void preorder(BinaryTreeNode *root,vector&v2)
{
	if(!root)
	{
		return;
	}
	v2.push_back(root->data);
	preorder(root->left,v2);
	preorder(root->right,v2);
}
void postorder(BinaryTreeNode *root,vector&v3)
{
	if(!root)
	{
		return;
	}
	postorder(root->left,v3);
	postorder(root->right,v3);
	v3.push_back(root->data);
}
vector> getTreeTraversal(BinaryTreeNode *root){
   vector>ans;
	vectorv1;
	vectorv2;
	vectorv3;
	
	inorder(root,v1);
	preorder(root,v2);
	postorder(root,v3);
	
	ans.push_back(v1);
	ans.push_back(v2);
	ans.push_back(v3);
	return ans;
	
}
