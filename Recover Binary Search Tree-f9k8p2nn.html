Recover Binary Search Tree

Brute: O(2n+nlogn) 
space:O(n)

 int i=0;
    void solve(vector&v,TreeNode* root)
    {
        if(!root)
        {
            return;
        }
        solve(v,root->left);
        v.push_back(root->val);
        solve(v,root->right);
    }
    void solve1(TreeNode* root,vector&v)
    {
        if(!root)
        {
            return;
        }
        solve1(root->left,v);
        if(root->val!=v[i])
        {
            root->val=v[i];
        }
        i++;
        solve1(root->right,v);
    }
