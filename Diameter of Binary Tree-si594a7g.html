Diameter of Binary Tree

Diameter is a longest path
defination :longest  path b/w any 2 nodes 
Path does't need to pass via root.


example :1 2 3 4 7 -1 -1 -1 -1 -1 -1
o/p:3

Brute approach: find each left and right height from each node 
BRUTE: time:O(n^n)

class Solution {
public:
    int maxi=-1;
    public:
int height(TreeNode* root)
{
	if(!root)
	{
		return 0;
	}
	int lh=height(root->left);
	int rh=height(root->right);
	return 1+max(lh,rh);
}
    int diameterOfBinaryTree(TreeNode* root) {
     
       if(!root)
	{
		return 0;
	}
	int lh=height(root->left);
	int rh=height(root->right);
	maxi=max(maxi,lh+rh);
	diameterOfBinaryTree(root->left);
	diameterOfBinaryTree(root->right);
	return maxi;
    }
};



OPTIMAL:time:O(n)

int height(TreeNode *root ,int &maxi)
{
	if(!root)
	{
		return 0;
	}
	int lh=height(root->left,maxi);
	int rh=height(root->right,maxi);
	maxi=max(maxi,lh+rh);
	return 1+max(lh,rh);
}
int diameterOfBinaryTree(TreeNode *root)
{
	int maxi=0;
	height(root,maxi);
	return maxi;
}
