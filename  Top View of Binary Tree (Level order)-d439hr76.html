o/p:8 4 2 1 3 7

approach : take set and queue (set stores)

#include<map> vector<int> getTopView(TreeNode<int> *root) { queue<pair<TreeNode<int>*,int>>q; vector<int>ans; if(!root) return ans; map<int,int>mp; q.push({root,0}); while(!q.empty()) { auto p=q.front(); q.pop(); TreeNode<int> * node=p.first; int index=p.second; if(mp.find(index)==mp.end()) { mp[index]=node->val; } if(node->left!=NULL) { q.push({node->left,index-1}); } if(node->right!=NULL) { q.push({node->right,index+1}); } } for(auto it:mp) { ans.push_back(it.second); } return ans; }
