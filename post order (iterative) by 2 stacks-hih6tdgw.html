post order (iterative) by 2 stacks

postorder=left right root

steps:1  makes 2 stacks and one vector 
2: push stack 1 to root and then find that st.top have left or right if have then push to that stack and push st.top to another stack .
3: while stack 2 not empty pop all elements and push to vector 
4: return vector.


  vector postorderTraversal(TreeNode* root) {
     stackst1;
     stackst2;
        vectorans;
        if(!root)
        {
            return ans;
        }
     st1.push(root);
        while(!st1.empty())
        {
            TreeNode* node=st1.top();
            st1.pop();
          
            if(node->left!=NULL)
            {
                st1.push(node->left);
            }
             if(node->right!=NULL)
            {
                st1.push(node->right);
            }
              st2.push(node);
        }
        while(!st2.empty())
        {
            TreeNode* node1=st2.top();
            st2.pop();
            ans.push_back(node1->val);
        }
        return ans;
    }


Input: root = [1,null,2,3]
Output: [3,2,1]
