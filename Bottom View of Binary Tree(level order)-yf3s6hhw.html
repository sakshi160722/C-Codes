Bottom View of Binary Tree(level order)

Bottom View of Binary Tree: we use level order traversal

class Solution {

  public:

    vector <int> bottomView(Node *root) {

    vector<int>ans;

if(root==NULL)

{

return ans;

}

map<int,int>mp;

queue<pair<Node *,int>>q;

q.push({root,0});

while(!q.empty())

{

auto p=q.front();

q.pop();

Node* node=p.first;

int index=p.second;

mp[index]=node->data;

if(node->left!=NULL)

{

q.push({node->left,index-1});

}

if(node->right!=NULL)

{

q.push({node->right,index+1});

}

 

}

for(auto it:mp)

{

ans.push_back(it.second);

}

   return ans;

    }

};
