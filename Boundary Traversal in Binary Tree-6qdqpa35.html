Boundary Traversal in Binary Tree

Print boundary elements for anti-clock wise (from left to right):

Approach:

  • Left boundary excluding leaf nodes
  • Leaf nodes (we use inorder traversal NOT level order because in level order suppose right subtree leafs are end before than left subtree leafs ).so on that level order doesn't works. 
  • Right boundary in reverse order excluding the leaf nodes

I/p:1 2 3 4 -1 5 -1 -1 -1 -1 -1
o/p:1 2 4 5 3

class Solution {

public:

 

    bool isleaf(Node *root)

    {

        return !root->left&& !root->right;

    }

   void  leftboun(Node *root, vector<int>&ans)

    {

        Node* cur=root->left;

        while(cur)

        {

            if(!isleaf(cur))

            {

                ans.push_back(cur->data);

            }

            if(cur->left)

            {

               cur=cur->left;

            }

            else{

                cur=cur->right;

            }

        }

    }

  void addleaves(Node *root,vector<int>&ans)

  {

      if(isleaf(root))

      {

          ans.push_back(root->data);

          return;

      }

      if(root->left)

      {

          addleaves(root->left,ans);

      }

      if(root->right)

      {

          addleaves(root->right,ans);

      }

  }

  void rightboun(Node *root,vector<int>&ans)

  {

     

      Node* cur=root->right;

       vector<int>v;

      while(cur)

      {

          if(!isleaf(cur))

          {

              v.push_back(cur->data);

          }

          if(cur->right)

          {

              cur=cur->right;

          }

          else{

               cur=cur->left;

          }

      }

      for(int i=v.size()-1;i>=0;i--)

      {

          ans.push_back(v[i]);

      }

  }

    vector <int> boundary(Node *root)

    {

        vector<int>ans;

       if(!root)

       {

           return ans;

       }

       if(!isleaf(root))

       {

           ans.push_back(root->data);

          

       }

       leftboun(root,ans);

       addleaves(root,ans);

       rightboun(root,ans);

       return ans;

    }

};
