Kth Smallest Element in a BST 

Brute:
Time:O(NlogN)
space:O(n)
----------------------------------  
void solve(TreeNode* root,vector&v)
    {
         if(!root)
        {
            return;
        }
        v.push_back(root->val);
        solve(root->left,v);
        solve(root->right,v);
    }
    int kthSmallest(TreeNode* root, int k) {
      vectorv;
       solve(root,v);
        sort(v.begin(),v.end());
       int t=v[k-1];
        return t;
            
    }


Better:------------------------------
time:O(n)
space:O(n)

int better we dont need to sort a vector bcz if we take inorder traversal so in inorder elements are automatically in sorted order   
void solve(Node* root,vector&v)
    {
        if(!root)
        {
            return;
        }
  solve(root->left,v);
        v.push_back(root->data);
  solve(root->right,v);
    }
    int kthLargest(Node *root, int K)
    {
        vectorv;
        solve(root,v);
        int t=v[K-1];
        if(K>v.size())
        {
            return -1;
        }
        return t;
    }

Optimal :----------------------------------
time:O(n)
space:O(1)
