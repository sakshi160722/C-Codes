preorder (iterative approach)

preorder= root left right 

vector preorderTraversal(TreeNode* root) {
      vectorans;
        if(!root)
        {
            return ans;
        }
        stackst;
        st.push(root);
        while(!st.empty())
        {
            TreeNode* node=st.top();
            st.pop();
            if(node->right!=NULL)
            {
                st.push(node->right);
            }
            if(node->left!=NULL)
            {
                st.push(node->left);
            }
            ans.push_back(node->val);
        }
        return ans;
    }
};
