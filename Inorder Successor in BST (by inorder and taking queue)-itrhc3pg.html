Inorder Successor in BST (by inorder and taking queue)

brute:-------------------
time:O(n)
space:O(n)
 void solve(queue&q,Node *root,Node *x)
  {
      if(!root)
      {
          return;
      }
      solve(q,root->left,x);
      q.push(root);
      solve(q,root->right,x);
  }
    // returns the inorder successor of the Node x in BST (rooted at 'root')
    Node * inOrderSuccessor(Node *root, Node *x)
    {
        if(root==NULL)
        {
            return NULL;
        }
       queueq;
        solve(q,root,x);
        Node* t=NULL;
        while(q.front()!=x)
        {
            q.pop();
        }
        q.pop();
        if(q.size()==0)
        {
            return t;
        }
        t=q.front();
       
        return t;
    }
