Zig-Zag or Spiral Traversal in Binary Tree(USE level order traversal)

in zigzag we have to print first row from left to right and second row from right to left and third from left to right as it's ......


use level order traversal and take one f integer initialise f=0; at starting then after check if (f%2!=0) means f==ODD so we have to reverse whole row and when f==EVEN so we don't need to reverse it and after we push vector into vector of vector at each row increase f++

Example :
Input: root = [3,9,20,null,null,15,7]
Output: [[3],[20,9],[15,7]]



class Solution {
public:
vector<vector> zigzagLevelOrder(TreeNode* root) {
int f=0;
vector<vector>ans;
if(!root)
{
return ans;
}
queue<TreeNode*>q;
q.push(root);
while(!q.empty())
{
int s=q.size();
vectorv;
for(int i=0;i<s;i++)
{
TreeNode* node=q.front();
q.pop();
v.push_back(node->val);
if(node->left!=NULL)
{
q.push(node->left);
}
if(node->right!=NULL)
{
q.push(node->right);
}

}
if(f%2!=0)
{
reverse(v.begin(),v.end());
}
f++;
ans.push_back(v);
}
return ans;
}
};
