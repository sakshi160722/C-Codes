Reverse Linked List

ListNode* reverseList(ListNode* head) {
      ListNode* cur=head;
        ListNode* node;
        ListNode* prev=NULL;           //1->2->3->4->5 
         while(cur!=NULL)
         {
           ListNode* node=cur->next; 
             cur->next=prev;
             prev=cur;
             cur=node;
            
         }                                 //  NULL<-1<-2
       return prev;
        
        
    }
