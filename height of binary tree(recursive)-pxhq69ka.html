height of binary tree(recursive)

 steps:1: int a variable store left count 
2: int b variable store right count 
3: 1+ because root is also added soo 1 for root  
4:return 1+max(a,b);

int maxDepth(TreeNode* root) {
      if(!root)
      {
          return 0;
      }
        int a=maxDepth(root->left);
        int b=maxDepth(root->right);
        return 1+max(a,b);
    }
