Vertical Order Traversal of Binary Tree (LEVEL order traversal )

Vertical Order Traversal of Binary Tree : vertical order traversal means draw a vertical lines in tree and print elements in vertical order.

  • for vertical lines write from top to bottom.
  • use LEVEL order traversal techniqueue

.........-2 -1 0 1 2...........

move ascending order of the levels and ascending order of top to bottom 

Data structure: take map with map and multiset.  map<int,map<int,multiset<int>>>mp;

                                                                      verticals    levels      multinodes

and take Queue data structure   Q(node,vertical,level)

 vector<vector<int>> verticalTraversal(TreeNode* root) {

        map<int,map<int,multiset<int>>>mp;

        queue<pair<TreeNode*,pair<int,int>>>q;

        q.push({root,{0,0}});

        while(!q.empty())

        {

          auto  n=q.front();

            q.pop();

            TreeNode* node=n.first;

            int x=n.second.first;

            int y=n.second.second;

            mp[x][y].insert(node->val);

            if(node->left!=NULL)

            {

                q.push({node->left,{x-1,y+1}});

            }

            if(node->right!=NULL)

            {

                q.push({node->right,{x+1,y+1}});

            }

        }

        vector<vector<int>>ans;

        for(auto p:mp)

        {

            vector<int>v;

            for(auto t:p.second)

            {

                v.insert(v.end(),t.second.begin(),t.second.end());

            }

            ans.push_back(v);

        }

        return ans;

    }
