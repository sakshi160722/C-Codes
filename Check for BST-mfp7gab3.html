bool isB(Node* root,int mini,int maxi)
    {
        if(!root)
        {
            return true;
        }
        if(root->data>=maxi || root->data<=mini)
        {
            return false;
        }
        return (isB(root->left,mini,root->data)&& isB(root->right,root->data,maxi));
    }
    bool isBST(Node* root) 
    {
     
      return isB(root,INT_MIN,INT_MAX);
    }
